\begin{frame}[fragile]{Structure générale d'une page web}
\begin{html}

<html lang="fr">
  <head>
    <meta charset="utf-8">
    <title>Titre</title>
  </head>
  <body>
    Bonjour !
  </body>
</html>
\end{html}

\begin{frame}[fragile]{Explication des balises}
  Des mots \underline{soulignés}, \textbf{gras} ou en \emph{italique}.

  \vfill

\begin{html}
<p>Des mots
<u>soulignés</u>,
<b>gras</b>,
ou en <em>italique</em>.
</p>
\end{html}
\end{frame}
