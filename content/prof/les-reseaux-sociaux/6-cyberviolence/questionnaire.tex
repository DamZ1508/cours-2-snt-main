\begin{titre}{2NDE}{Seconde - SNT}
\centering
{\Large \textsc{\textbf{Thème : Réseaux sociaux --- Harcèlement}}}
\end{titre}

{\em Séance inspirée du \enquote{Cahier d'activités - Secondaire : Outils et séquences pédagogiques}}\\
{\em \scriptsize \url{https://www.nonauharcelement.education.gouv.fr/ressources/cahier-dactivites-second-degre-outils-et-sequences-pedagogiques/}}

\begin{enumerate}
  \item Quels sont les acteurs du harcèlement ?
  \item Pourquoi peut-on parler de harcèlement dans les deux vidéos ?
  \item Quelles sont les conséquences du harcèlement sur la victime ?
  \item Qu'est-ce qui peut empêcher la victime de se défendre ? Que peut ressentir la victime ?
  \item Quelles autres formes ou pratiques de harcèlement connaissez-vous ?
  \item Quelles sont les spécificités du cyberharcèlement ? Pourquoi peut-on dire que l'utilisation des multimédias facilite et accentue le phénomène de harcèlement ?
  \item A-t-on le droit de prendre une photo d'un camarade et de la diffuser sur les réseaux sociaux ? A-t-on le droit de diffuser une photo d'un camarade sans son accord ?
  \item A-t-on le droit de faire une blague sexiste, homophobe, ou raciste en public ?
  \item Pourquoi est-il difficile, en tant que témoin, d'agir ? En quoi la réaction du témoin change-t-elle la situation ? Ne pas intervenir, est-ce consentir ?
  \item Que peuvent faire les différents acteurs pour arrêter cette situation de harcèlement ?
\end{enumerate}
