\begin{definition}
Tout point à la surface de la Terre est déterminé par \textbf{ses coordonnées géographiques} et par son \textbf{altitude} (élévation par rapport au niveau de la mer).
\saut 
L'ensemble de ces notions, auquel on ajoute le centre de la Terre, est utilisé comme système de référence mondial pour déterminer les positions sur la Terre par les systèmes GPS, dont nous parlerons dans la séance 2.
\end{definition}

\begin{definition}
Les coordonnées géographiques comprennent :
\begin{itemize}
  \item \textbf{la latitude} : angle par rapport à \trou{l'équateur} ;
    \item \textbf{la longitude} : angle par rapport au \trou{méridien} de référence (celui de Greenwitch) ;
\end{itemize}
\end{definition}

\exo[Indiquer des coordonnées géographiques à partir d'une carte]

Indiquer la latitude et la longitude des villes placées sur la carte ci-dessous \textit{(en arrondissant au degré près)} :
\begin{center}
	\includegraphics[scale=0.4]{ex1.eps}
\end{center}

\sautpage

\begin{prop}
Ces angles peuvent être donnés : 
\begin{itemize}
	\item en \textbf{degrés décimaux} \textit{(notation décimale)} (degrés et dixièmes, centièmes\ldots{} de degrés) ;
	\item en \textbf{degrés-minutes-secondes} \textit{(notation sexadécimale - DMS)} (avec : $1\degre = 60' = \np{3600}''$)
\end{itemize}
\end{prop}

\begin{remarque}
\begin{itemize}
	\item Une latitude positive donne la direction Nord par rapport à l'équateur et une latitude négative donne Sud.
	\item Une longitude positive donne la direction Est et une longitude négative donne Ouest.
\end{itemize}
\end{remarque}

\begin{exemple}
Par exemple, les coordonnées de la salle de SNT peuvent être écrites de différentes manières :
\begin{itemize}
  \item degrés décimaux : $\coord{46,840381\degre}{-0,485806\degre}$, soit :
    \begin{itemize}
      \item 46,840381 degrés au Nord de l'équateur ;
      \item 0,485806 degrés à l'ouest du méridien de Greenwitch.
    \end{itemize}

  \item degrés-minutes-secondes : $\coord{46\degre 50' 25.37'' ~ N}{0\degre 29' 8,9'' ~ O}$, soit :
    \begin{itemize}
      \item 46 degrés 50 minutes 25,37 secondes au nord de l'équateur ;
      \item 0 degrés 29 minutes 8,9 secondes à l'ouest du méridien de Greenwitch..
    \end{itemize}
\end{itemize}
\end{exemple}

\begin{methode}
\begin{enumerate}
	\item \textbf{Passer des degrés sexagésimaux aux degrés décimaux :}

Les coordonnées de la salle de SNT sont $\coord{46\degre 50' 25.37'' ~ N}{0\degre 29' 8,9'' ~ O}$ :
\begin{itemize}
	\item 60' = 1° donc 50'=50/60 ° et 3600' = 1° donc 25,37''=25,37/3600°
	\item Latitude en degrés décimaux : $46 + \dfrac{50}{60} + \dfrac{25,37}{3600} \approx 46,840381 \degre$
	\item Longitude en degrés décimaux : $0 + \dfrac{29}{60} + \dfrac{8,9}{3600} \approx 20,485806 \degre$
\end{itemize}

	\item \textbf{Passer des degrés décimaux aux degrés sexagésimaux :}

Les coordonnées de la salle de SNT sont $\coord{46,840381\degre}{-0,485806\degre}$ en degrés décimaux.
\vspace*{-0.25cm}\begin{multicols}{2}
\begin{itemize}
	\item Latitude en degrés sexagésimaux :
	\begin{itemize}
		\item Nord et on garde 46 pour les degrés ;
		\item $0,840381 \times 60 = 50,42286$ : on garde 50' ;
		\item $0,42286 \times 60 \approx 25,37$ et on garde 25,37''
	\end{itemize}
\columnbreak
	\item Longitude en degrés sexagésimaux :
	\begin{itemize}
		\item Le signe négatif donne Ouest et on garde 0 pour les degrés ;
		\item $0,485806 \times 60 = 29,14836$ : on garde 29' ;
		\item $0,14836 \times 60 \approx 8,9$ et on garde 8,9''
	\end{itemize}
\end{itemize}
\end{multicols}
\end{enumerate}
\end{methode}

\exo[Passer du système sexagésimal au système décimal et inversement]
\saut \vspace*{-0.4cm}
\begin{enumerate}
	\item On donne les coordonnées suivantes en degrés décimaux : $\coord{40,689253 \degre}{-74,044548 \degre}$. 
	
	Convertir ces coordonnées en DMS puis indiquer le lieu désigné.
	\item On donne les coordonnées en DMS suivantes : $\coord{55 \degre 45' 13''~ N}{37 \degre 37' 17''~ E}$. 
	
	Convertir ces coordonnées en coordonnées décimales puis indiquer le lieu désigné.
\end{enumerate}

\exo
\saut \vspace*{-0.4cm}
\begin{enumerate}
	\item $\coord{40,689253 \degre}{-74,044548 \degre}$
	\begin{itemize}
		\item Latitude : $0,689253 \times 60 = 41,35518 \rightarrow 41'$ et $0,35518 \times 60 = 21,3108 \rightarrow 21''$
		\item Longitude : $0,044548 \times 60 = 2,67288 \rightarrow 2' et 0,67288 \times 60 = 40,3728 \rightarrow 40''$
		\item Les coordonnées sont $\coord{40\degre~ 41' ~ 21'' ~N}{74\degre ~2'~40''~ O}$. Ce sont les coordonnées de la statue de la liberté.
	\end{itemize}
	\item $\coord{55 \degre 45' 13''~ N}{37 \degre 37' 17''~ E}$
	\begin{itemize}
		\item Latitude : 55 + 45/60 + 13/3600 = 55,753611° (positif car Nord)
		\item Longitude : 37+ 37/60 + 17/3600 = 37,621389° (positif car Est)
		\item Les coordonnées sont $\coord{55,753611\degre}{37,621389\degre}$. Ce sont les coordonnées de la Place Rouge à Moscou.
	\end{itemize}
\end{enumerate}
